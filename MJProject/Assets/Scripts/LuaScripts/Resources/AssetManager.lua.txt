local LuaReader = require "LuaReader"
local protobuf = require "protobuf"
local util = require "xlua.util"
local function AsyncYieldReturn(to_yield, cb)
    Player.self:YieldAndCallback(to_yield, cb)
end
local yield_return = util.async_to_sync(AsyncYieldReturn)

AssetManager = {}
local this = AssetManager

---------------------------------------------------------------------------
--成员变量
---------------------------------------------------------------------------
local _loaded = false
local _copyed = false

---------------------------------------------------------------------------
--加载资源
---------------------------------------------------------------------------
function AssetManager.Load(assetFilePath)
	if _loaded then return true end
	-----------------------------------------------
	_loaded = LuaReader.Load(assetFilePath)
	return _loaded
end

---------------------------------------------------------------------------
--复制资源
---------------------------------------------------------------------------
function AssetManager.CopyAndroidRes( ... )
	-- body
	--android load
	if _copyed then return true end
	_copyed = true
	local folder = CS.System.IO.DirectoryInfo(CS.LuaCommon.resultPath)
	local co = coroutine.create(function()
		for k,v in pairs(folder:GetFileSystemInfos("*")) do
			--print(v.Name)
			for j,c in pairs(v:GetFiles("*")) do
		        local www = CS.UnityEngine.WWW(c.FullName)
		        yield_return(www)
		        if not www.error then
		        	local sw
		            local t = CS.System.IO.FileInfo(CS.UnityEngine.Application.persistentDataPath.."/Asset/Asset/"..v.Name.."/"..c.Name)
		            if not t.Exists then
		            	sw = t:CreateText()
		            else
		            	sw = t:AppendText()
		            end
		            sw.WriteLine(www.text)
				    sw.Close()
				    sw.Dispose()
		        else
		            CS.UnityEngine.Debug.LogWarning('error:'..www.error)
		        end
			end
		end
    end)
    coroutine.resume(co)
	return _copyed
end

---------------------------------------------------------------------------
--打印数据
---------------------------------------------------------------------------
function AssetManager.PrintMessage(message)
	if type(message) == "table" then
		for k, v in pairs(message) do
			if type(v) ~= "table" then
				print (k, v)
			else 
				AssetManager.PrintMessage(v)
			end
		end
	else
		print (message)
	end
end

function AssetManager.PrintDecodeMessage(message,message_type)
	-- body
	local messageTable = {}
	for k, value in pairs(message) do
		local tempMessage = protobuf.decode(message_type, value)
		messageTable[k]  = tempMessage
	end
	return messageTable
end
