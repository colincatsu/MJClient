PaiAnimation = {}
local AnimationCurve = CS.UnityEngine.AnimationCurve
local Animation = CS.UnityEngine.Animation
local AnimationClip = CS.UnityEngine.AnimationClip
local Transform = CS.UnityEngine.Transform

local pai_in_up = {--抬起
	time = 0.5,
	position_offset_x = 1,
	position_offset_y = 0.03619,
	position_offset_z = 0.03619,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}
local pai_in_sort = {--插入移动
	time = 1,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = -20,
}
local pai_in_down = {--落下
	time = 1.5,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}
local pai_shift = {--其他牌横移
	time = 1,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}

function PaiAnimation.SetPaiInSort(self,object,toPositionX)
	-- body
	local myAnimation = object:GetComponent("Animation")
	if self:IsNull(myAnimation) == true then
		myAnimation = object:AddComponent(typeof(Animation))
	end
	local myAnimationClip = AnimationClip()
	myAnimationClip.legacy = true
	--
	curvePostionX.AddKey(0f,transform.localPosition.x);
		curvePostionX.AddKey(0.2f,0f);
		myAnimationClip.SetCurve("", typeof(Transform), "localPosition.x", curvePostionX)
	--
	--抬起
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.x", AnimationCurve.EaseInOut(0,object.transform.localPosition.x,pai_in_up.time,object.transform.localPosition.x))
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.y", AnimationCurve.EaseInOut(0,object.transform.localPosition.y,pai_in_up.time,object.transform.localPosition.y+pai_in_up.position_offset_y*math.cos(math.rad(object.transform.localEulerAngles.x))))
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.z", AnimationCurve.EaseInOut(0,object.transform.localPosition.z,pai_in_up.time,object.transform.localPosition.z+pai_in_up.position_offset_z*math.sin(math.rad(object.transform.localEulerAngles.x))))
	--横移
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.x", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localPosition.x,pai_in_sort.time,toPositionX))
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.y", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localPosition.y+pai_in_up.position_offset_y*math.cos(math.rad(object.transform.localEulerAngles.x)),pai_in_sort.time,object.transform.localPosition.y+pai_in_up.position_offset_y*math.cos(math.rad(object.transform.localEulerAngles.x))))
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.z", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localPosition.z+pai_in_up.position_offset_z*math.sin(math.rad(object.transform.localEulerAngles.x)),pai_in_sort.time,object.transform.localPosition.z+pai_in_up.position_offset_z*math.sin(math.rad(object.transform.localEulerAngles.x))))
	--横移转角
	myAnimationClip:SetCurve("", typeof(Transform), "localEulerAngles.x", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localEulerAngles.x,pai_in_sort.time,object.transform.localEulerAngles.x))
	myAnimationClip:SetCurve("", typeof(Transform), "localEulerAngles.y", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localEulerAngles.y,pai_in_sort.time,object.transform.localEulerAngles.y))
	myAnimationClip:SetCurve("", typeof(Transform), "localEulerAngles.z", AnimationCurve.EaseInOut(pai_in_up.time,object.transform.localEulerAngles.z,pai_in_sort.time,object.transform.localEulerAngles.z+pai_in_sort.rotation_offset_z))

	myAnimation:AddClip(myAnimationClip, "Up")
	myAnimation:Play("Up")
end



return PaiAnimation
