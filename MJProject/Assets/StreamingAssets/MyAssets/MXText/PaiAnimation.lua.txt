PaiAnimation = {}
local AnimationCurve = CS.UnityEngine.AnimationCurve
local Animation = CS.UnityEngine.Animation
local AnimationClip = CS.UnityEngine.AnimationClip
local Transform = CS.UnityEngine.Transform

local pai_in_up = {--抬起
	time = 0.5,
	position_offset_x = 1,
	position_offset_y = 0.03619,
	position_offset_z = 0.03619,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}
local pai_in_sort = {--插入移动
	time = 1,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = -20,
}
local pai_in_down = {--落下
	time = 1.5,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}
local pai_shift = {--其他牌横移
	time = 1,
	position_offset_x = 1,
	position_offset_y = 1,
	position_offset_z = 1,
	rotation_offset_x = 1,
	rotation_offset_y = 1,
	rotation_offset_z = 1,
}

function PaiAnimation.SetPaiInSort(self,object,toPositionX)
	-- body
	local myAnimation = object:GetComponent("Animation")
	if self:IsNull(myAnimation) == true then
		myAnimation = object:AddComponent(typeof(Animation))
	end
	local myAnimationClip = AnimationClip()
	local curvePostionX = AnimationCurve()
	local curvePostionY = AnimationCurve()
	local curvePostionZ = AnimationCurve()
	local curveRotationX = AnimationCurve()
	local curveRotationY = AnimationCurve()
	local curveRotationZ = AnimationCurve()

	myAnimationClip.legacy = true
	--抬起
	curvePostionX:AddKey(0,object.transform.localPosition.x)
	curvePostionX:AddKey(pai_in_up.time,object.transform.localPosition.x)
	curvePostionY:AddKey(0,object.transform.localPosition.y)
	curvePostionY:AddKey(pai_in_up.time,object.transform.localPosition.y+pai_in_up.position_offset_y*math.cos(math.rad(object.transform.localEulerAngles.x)))
	curvePostionZ:AddKey(0,object.transform.localPosition.z)
	curvePostionZ:AddKey(pai_in_up.time,object.transform.localPosition.z+pai_in_up.position_offset_z*math.sin(math.rad(object.transform.localEulerAngles.x)))
	--横移
	curvePostionX:AddKey(pai_in_sort.time,toPositionX)
	--横移转角
	curveRotationZ:AddKey(0,0)
	curveRotationZ:AddKey(pai_in_up.time,pai_in_sort.rotation_offset_z)

	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.x", curvePostionX)
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.y", curvePostionY)
	myAnimationClip:SetCurve("", typeof(Transform), "localPosition.x", curvePostionZ)
	myAnimationClip:SetCurve("", typeof(Transform), "localEulerAngles.z", curveRotationZ)

	myAnimation:AddClip(myAnimationClip, "Up")
	myAnimation:Play("Up")
end



return PaiAnimation
